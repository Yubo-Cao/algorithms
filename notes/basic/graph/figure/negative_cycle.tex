\documentclass[border=0.5cm]{standalone}
\usepackage{tikz}
\usetikzlibrary{graphs, quotes}

\begin{document}
\def\sz{0.5cm}
% draw a graph with a negative cycle
\begin{tikzpicture}
    \graph[
        edge quotes={fill=white,inner sep=1pt},
        nodes={circle, draw, minimum size=\sz, inner sep=0pt},
        grow right=2cm,
    ]
    {
        1 ->["3"] 2 ->[red, "5"] 3 ->["2"] 5;
        3 ->[red, "-2"] 4;
        4 ->[red, "-4"] 2;
    };
\end{tikzpicture}
\end{document}
